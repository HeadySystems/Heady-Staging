API Endpoints

RESTful endpoints powered by HCFullPipeline v7. Every request traverses our 21-stage sacred geometry pipeline.

GET /v1/health

Returns the current health status, coherence score, and version of the HeadyIO service cluster.

{
  "status": "healthy",
  "coherenceScore": 0.927,
  "version": "4.1.0",
  "pipeline": "HCFullPipeline-v7",
  "uptime": 99.97,
  "activeNodes": 21,
  "latency": {
    "p50": 3.2,
    "p95": 8.1,
    "p99": 14.4
  }
}
POST /v1/pipeline/execute

Execute a request through the full 21-stage HCFullPipeline. Supports FAST_PATH (7 stages), FULL_PATH (21 stages), ARENA_PATH (9 stages), and LEARNING_PATH (7 stages).

    "embeddings": [0.123, -0.456, "...384D"]
  },
  "payload": {
    "query": "Analyze system coherence"
  }
}

// Response 200 OK
{
  "pipelineId": "pipe_f8e7d6c5",
  "stagesCompleted": 21,
  "coherenceScore": 0.882,
  "result": { /* ... */ }
}

Client Libraries

Official SDKs

First-class SDK support for Node.js, Python, and Go. Each SDK wraps the full HCFullPipeline with type-safe interfaces.

heady-sdk-node Node.js 
import { HeadyIO } from '@heady/sdk-node';

// Initialize the HeadyIO client with sacred geometry defaults
const heady = new HeadyIO({
  apiKey: process.env.HEADY_API_KEY,
  baseUrl: 'https://api.headyio.com/v1',
  pipeline: 'FULL_PATH',
  timeout: 8000,
  retryStrategy: {
    maxAttempts: 5,
    backoff: 'phi-exponential',  // PHI^attempt * base
    jitter: 0.382              // +/- 38.2% (1 - PSI)
  }
});

// Execute a pipeline request
const result = await heady.pipeline.execute({
  intent: 'classify_and_route',
  context: {
    channelId: 'web-dashboard',
    userId: 'usr_a1b2c3d4'
  },
  payload: {
    query: 'Analyze system coherence metrics'
  }
});

console.log(result.coherenceScore); // 0.882
console.log(result.stagesCompleted); // 21

// Subscribe to real-time events via WebSocket
const stream = heady.events.subscribe({
  channels: ['pipeline.*', 'nodes.health'],
  filter: {
    coherenceMin: 0.691
  }
});

stream.on('pipeline.complete', (event) => {
  console.log(`Pipeline ${event.pipelineId} completed`);
});

stream.on('nodes.health', (event) => {
  console.log(`Node ${event.nodeId}: ${event.status}`);
});
heady-sdk-python Python 
import os
from heady import HeadyIO, PipelineConfig, RetryStrategy

# Initialize the HeadyIO client
client = HeadyIO(
    api_key=os.environ["HEADY_API_KEY"],
    base_url="https://api.headyio.com/v1",
    pipeline=PipelineConfig(
        variant="FULL_PATH",
        stages=21
    ),
    retry=RetryStrategy(
        max_attempts=5,
        backoff="phi-exponential",
        jitter=0.382
    )
)

# Execute pipeline with async context manager
async with client as heady:
    result = await heady.pipeline.execute(
        intent="classify_and_route",
        context={
            "channel_id": "web-dashboard",
            "user_id": "usr_a1b2c3d4",
        },
        payload={
            "query": "Analyze system coherence metrics"
        }
    )

    print(f"Coherence: {result.coherence_score}")  # 0.882
    print(f"Stages: {result.stages_completed}")    # 21

    # Fetch topology
    nodes = await heady.nodes.list()
    for ring, members in nodes.topology.items():
        print(f"{ring}: {', '.join(members)}")

    # Stream events with SSE
    async for event in heady.events.stream(
        channels=["pipeline.*"],
        coherence_min=0.691
    ):
        print(f"Event: {event.type} | {event.data}")
heady-sdk-go Go 
package main

import (
    "context"
    "fmt"
    "log"
    "os"

    heady "github.com/heady-systems/sdk-go"
)

func main() {
    // Initialize HeadyIO client with phi-scaled retry
    client, err := heady.NewClient(heady.Config{
        APIKey:   os.Getenv("HEADY_API_KEY"),
        BaseURL:  "https://api.headyio.com/v1",
        Pipeline: heady.FullPath,
        Retry: heady.RetryConfig{
            MaxAttempts: 5,
            Backoff:     heady.PhiExponential,
            Jitter:      0.382,
        },
    })
    if err != nil {
        log.Fatal(err)
    }
    defer client.Close()

    ctx := context.Background()

    // Execute pipeline
    result, err := client.Pipeline.Execute(ctx, &heady.PipelineRequest{
        Intent: "classify_and_route",
        Context: heady.RequestContext{
            ChannelID: "web-dashboard",
            UserID:    "usr_a1b2c3d4",
        },
        Payload: map[string]interface{}{
            "query": "Analyze system coherence",
        },
    })
    if err != nil {
        log.Fatal(err)
    }

    fmt.Printf("Coherence: %.3f\n", result.CoherenceScore)
    fmt.Printf("Stages: %d\n", result.StagesCompleted)

    // Stream real-time events
    events, err := client.Events.Subscribe(ctx, &heady.SubscribeConfig{
        Channels:     []string{"pipeline.*"},
        CoherenceMin: 0.691,
    })
    if err != nil {
        log.Fatal(err)
    }

    for event := range events {
        fmt.Printf("Event: %s | %v\n", event.Type, event.Data)
    }
}
Security

Authentication & OAuth 2.1

Zero-trust security enforced at every node. PKCE-required OAuth 2.1 with phi-derived token rotation.

1

Register Application

Create your app in the HeadyIO developer console. Receive your client_id and configure redirect URIs.

2

PKCE Challenge

Generate a code_verifier and code_challenge. OAuth 2.1 mandates PKCE for all grant types.

3

Authorization

Redirect user to /oauth/authorize with scope, state, and PKCE parameters.

4

Token Exchange

Exchange authorization code for access and refresh tokens. Tokens rotate on a phi-scaled schedule.

OAuth 2.1 + PKCE Flow cURL 
# Step 1: Authorization Request (browser redirect)
GET https://auth.headyio.com/oauth/authorize
  ?response_type=code
  &client_id=hdy_client_a1b2c3
  &redirect_uri=https://yourapp.com/callback
  &scope=pipeline:execute nodes:read events:subscribe
  &state=random_csrf_token
  &code_challenge=E9Melhoa2OwvFrEMTJguCHaoeK1t8URWbuGJSstw-cM
  &code_challenge_method=S256

# Step 2: Token Exchange
curl -X POST https://auth.headyio.com/oauth/token \
  -H "Content-Type: application/json" \

    "grant_type": "authorization_code",
    "code": "auth_code_from_callback",
    "client_id": "hdy_client_a1b2c3",
    "redirect_uri": "https://yourapp.com/callback",
    "code_verifier": "your_original_code_verifier"
  }'

# Response

# Step 3: Authenticated API Request

Streaming

Real-time Events

Live data streams via WebSocket and Server-Sent Events. Subscribe to pipeline completions, node health, and system coherence in real time.

⚡

WebSocket

Full-duplex communication for bidirectional real-time streaming. Subscribe to channels, send commands, and receive events with sub-millisecond delivery.

// Connect to WebSocket endpoint
const ws = new WebSocket(
  'wss://ws.headyio.com/v1/events'
);

ws.onopen = () => {
  ws.send(JSON.stringify({
    type: 'subscribe',
    channels: [
      'pipeline.complete',
      'nodes.health',
      'coherence.drift'
    ],
    auth: 'hdy_at_eyJhbG...'
  }));
};

ws.onmessage = (event) => {
  const data = JSON.parse(event.data);
  // { channel, type, payload, ts }
  console.log(data.channel, data.payload);
};
📡

Server-Sent Events

Lightweight unidirectional streaming over HTTP. Automatic reconnection with phi-scaled backoff. Ideal for dashboards and monitoring interfaces.

// Connect to SSE endpoint
const source = new EventSource(
  'https://api.headyio.com/v1/events/stream'
  + '?channels=pipeline.*,nodes.*'
  + '&token=hdy_at_eyJhbG...'
);

source.addEventListener(
  'pipeline.complete',
  (event) => {
    const data = JSON.parse(event.data);
    console.log(
      `Pipeline ${data.id} done`,
      `Coherence: ${data.coherence}`
    );
  }
);

source.addEventListener(
  'nodes.health',
  (event) => {
    const data = JSON.parse(event.data);
    console.log(
      `${data.node}: ${data.status}`
    );
  }
);
